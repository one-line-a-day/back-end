- USERS ROUTES -

Note: Authorization tokens last 60minutes.

POST - `/api/users/register` -

register an account

required body: {username, email, password}

response: {message, id, token, username, name}

POST - `/api/users/login` -

login and get a token

required body: {username, password}

response: {message, id, token, username, name}

note: username key will work with both matching username or email value for account.

PATCH - `/api/users/:id` -

update a users info

required headers: {Authorization: token}

required body: {(optional: username, password, name, email)}

DELETE = `/api/users/:id` -

delete user and corresponding lines

required headers: {Authorization: token}

response: {message}

- LINES ROUTES -

GET - `/api/lines` -

get all lines of user on token

required headers: {Authorization: token}

Response: [ lines ]

GET - `/api/lines/YYYY-MM-DD` -

get line by date (please use leading 0s for days/months 1-9

required headers: {Authorization: token}

Response: {line, date, id, img_url}

GET - `/api/lines/id/:id` -

get line by date (please use leading 0s for days/months 1-9

required headers: {Authorization: token}

Response: {line, date, id, img_url}

GET - `/api/lines/month/MM/year/YYYY`-

get array of lines matching the month and year

required headers: {Authorization: token}

Response: [ lines ]

POST - `/api/lines` -

add new line for user on token

required headers: {Authorization: token}

required body: {line, date: 'YYYY-MM-DD' }

response: {message, id}

PATCH - `/api/lines/:id` -

update a line's information

required headers: {Authoerization: token}

required body: {(optional: line, date, img_url)}

response: {updatedLine}

DELETE - `/api/lines/:id` -

deletes line of specified id

required headers: required headers: {Authoerization: token}

response: {message}
